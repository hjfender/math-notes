\section{Fundamental Theorem of Algebra}\label{fundamentalthmofalgebra}

\begin{theorem}
If $p(z) = z^n + a_{n-1}z^{n-1}+ \cdots + a_1z + a_0$ is a polynomial with complex coefficients, then there is a complex number $z_0$ with $p(z_0) = 0$.
\end{theorem}
